74// Main entry point for any SMACC state machine
75// It instantiates and starts the specified state machine type
76// it uses two threads: a new thread and the current one.
77// The created thread is for the state machine process
78// it locks the current thread to handle events of the state machine
79template <typename StateMachineType>
80void run()
81{
82 // create the asynchronous state machine scheduler
83 SmaccFifoScheduler scheduler1(true);
84
85 // create the signalDetector component
86 SignalDetector signalDetector(&scheduler1);
87
88 // create the asynchronous state machine processor
89 SmaccFifoScheduler::processor_handle sm =
90 scheduler1.create_processor<StateMachineType>(&signalDetector);
91
92 // initialize the asynchronous state machine processor
93 signalDetector.setProcessorHandle(sm);
94
95 scheduler1.initiate_processor(sm);
96
97 //create a thread for the asynchronous state machine processor execution
98 boost::thread otherThread(boost::bind(&sc::fifo_scheduler<>::operator(), &scheduler1, 0));
99
100 // use the main thread for the signal detector component (waiting actionclient requests)
101 signalDetector.pollingLoop();
102 scheduler1.terminate();
103 otherThread.join();
104}
