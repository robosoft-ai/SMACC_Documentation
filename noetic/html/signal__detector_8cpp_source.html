42void SignalDetector::findUpdatableClients()
43{
44 this->updatableClients_.clear();
45 for (auto pair : this->smaccStateMachine_->getOrthogonals())
46 {
47 auto &orthogonal = pair.second;
48 auto &clients = orthogonal->getClients();
49
50 for (auto &client : clients)
51 {
52 // updatable client components
53 auto updatableClient = dynamic_cast<ISmaccUpdatable *>(client.get());
54
55 if (updatableClient != nullptr)
56 {
57 ROS_DEBUG_STREAM("Adding updatable client: " << demangleType(typeid(updatableClient)));
58 this->updatableClients_.push_back(updatableClient);
59 }
60
61 // updatable client components
62 std::vector<std::shared_ptr<ISmaccComponent>> components;
63 client->getComponents(components);
64 for (auto &componententry : components)
65 {
66 auto updatableComponent = dynamic_cast<ISmaccUpdatable *>(componententry.get());
67 if (updatableComponent != nullptr)
68 {
69 ROS_DEBUG_STREAM("Adding updatable component: " << demangleType(typeid(*updatableComponent)));
70 this->updatableClients_.push_back(updatableComponent);
71 }
72 }
73 }
74 }

176void SignalDetector::pollOnce()
177{
178 // precondition: smaccStateMachine_ != nullptr
179
180 try
181 {
182 smaccStateMachine_->lockStateMachine("update behaviors");
183
184 this->findUpdatableClients();
185 ROS_DEBUG_STREAM("updatable clients: " << this->updatableClients_.size());
186
187 if (this->updatableClients_.size())
188 {
189 for (auto *updatableClient : this->updatableClients_)
190 {
191 ROS_DEBUG_STREAM("[PollOnce] update client call: " << demangleType(typeid(updatableClient)));
192 updatableClient->executeUpdate();
193 }
194 }
195
196 // STATE UPDATABLE ELEMENTS
197 if (this->smaccStateMachine_->stateMachineCurrentAction != StateMachineInternalAction::TRANSITIONING &&
198 this->smaccStateMachine_->stateMachineCurrentAction != StateMachineInternalAction::STATE_CONFIGURING &&
199 this->smaccStateMachine_->stateMachineCurrentAction != StateMachineInternalAction::STATE_EXITING &&
200 this->smaccStateMachine_->stateMachineCurrentAction != StateMachineInternalAction::STATE_ENTERING)
201 {
202 // we do not update updatable elements during trasitioning or configuration of states
203 long currentStateIndex = smaccStateMachine_->getCurrentStateCounter();
204
205 ROS_DEBUG_STREAM("[PollOnce] update behaviors. checking current state");
206
207 auto currentState = smaccStateMachine_->getCurrentState();
208 if (currentState != nullptr)
209 {
210 ROS_DEBUG_STREAM("[PollOnce] current state: " << currentStateIndex);
211 ROS_DEBUG_STREAM("[PollOnce] last state: " << this->lastState_);
212
213 if (currentStateIndex != 0)
214 {
215 if (currentStateIndex != this->lastState_)
216 {
217 ROS_DEBUG_STREAM("[PollOnce] detected new state, refreshing updatable client behavior table");
218 // we are in a new state, refresh the updatable client behaviors table
219 this->lastState_ = currentStateIndex;
220 this->findUpdatableStateElements(currentState);
221 }
222
223 ROS_DEBUG_STREAM("updatable state elements: " << this->updatableStateElements_.size());
224 for (auto *udpatableStateElement : this->updatableStateElements_)
225 {
226 ROS_DEBUG_STREAM("pollOnce update client behavior call: " << demangleType(typeid(*udpatableStateElement)));
227 udpatableStateElement->executeUpdate();
228 }
229 }
230 }
231 }
232 }
233 catch (std::exception &ex)
234 {
235 ROS_ERROR("Exception during Signal Detector update loop. %s", ex.what());
236 }
237
238 smaccStateMachine_->unlockStateMachine("update behaviors");
239}

246void SignalDetector::pollingLoop()
247{
248 // ros::NodeHandle nh("~"); // use node name as root of the parameter server
249
250 ros::NodeHandle _;
251 ros::Rate r0(20);
252 while (!initialized_)
253 {
254 r0.sleep();
255 }
256
257 ros::NodeHandle nh(smacc::utils::cleanShortTypeName(typeid(*this->smaccStateMachine_)));
258
259 if (!nh.getParam("signal_detector_loop_freq", this->loop_rate_hz))
260 {
261 ROS_WARN("Signal detector frequency (ros param ~signal_detector_loop_freq) was not set, using default frequency: "
262 "%lf",
263 this->loop_rate_hz);
264 }
265 else
266 {
267 ROS_WARN("Signal detector frequency (ros param ~signal_detector_loop_freq): %lf", this->loop_rate_hz);
268 }
269
270 nh.setParam("signal_detector_loop_freq", this->loop_rate_hz);
271
272 ROS_INFO_STREAM("[SignalDetector] loop rate hz:" << loop_rate_hz);
273
274 ros::Rate r(loop_rate_hz);
275 while (ros::ok() && !end_)
276 {
277 ROS_INFO_STREAM_THROTTLE(10, "[SignalDetector] heartbeat");
278 pollOnce();
279 ros::spinOnce();
280 r.sleep();
281 }
282}
