28def build_deb_package(
29 workspace_source_folder, package_name, packagepath, ubuntu_version, ros_distro, already_visited
30):
31 os.chdir(workspace_source_folder)
32 print("------------------------------------------------------------")
33 print("Working folder: " + str(os.getcwd()))
34 print("Building debian package: " + str(package_name))
35 cmd = (
36 "bloom-generate rosdebian --os-name ubuntu --os-version "
37 + str(ubuntu_version)
38 + " --ros-distro "
39 + str(ros_distro)
40 + " "
41 + str(packagepath)
42 )
43
44 # finding actual source package directory of the package in the workspace
45 develpackagefolder = None
46 for root, dirs, files in os.walk(workspace_source_folder, topdown=False):
47 if "package.xml" in files and package_name == root.split("/")[-1] and "debian" not in root:
48 print(root)
49 develpackagefolder = root
50 break
51
52 if develpackagefolder is None:
53 print("ERROR: package " + str(package_name) + " not found in workspace")
54
55 localpackagepath = develpackagefolder
56 print(f"local package path: {localpackagepath} ")
57
58 # cleaning possible previous wrong builds
59 if os.path.exists(os.path.join(localpackagepath, "debian")):
60 shutil.rmtree(os.path.join(localpackagepath, "debian"))
61
62 # executing bloom to generate debian-fakeroot build data
63 print(cmd)
64 bloomprocess = subprocess.Popen(cmd, shell=True, stdout=subprocess.PIPE)
65 bloomprocess.wait()
66 print("bloom process finished: " + str(bloomprocess.returncode))
67
68 for line in bloomprocess.stdout:
69 print(str(line))
70
71 print("------------------------------------------------------------")
72 os.chdir(workspace_source_folder)
73 print(" - Local package path: " + localpackagepath)
74 print(" - Workspace directory:")
75 print(os.getcwd())
76 print(" - list localpackagepath:")
77 print(os.listdir(localpackagepath))
78
79 # copy the debian folder generated by bloom inside the package
80 shutil.move("debian", os.path.join(localpackagepath, "debian"))
81
82 os.chdir(localpackagepath)
83 fakerootprocess = subprocess.Popen("fakeroot debian/rules binary", shell=True)
84 (result, err) = fakerootprocess.communicate()
85
86 print("--------------------")
87 os.chdir(workspace_source_folder)
88
89 firstregexstr = ".*ros-" + ros_distro + r"-.*\.deb$"
90 regexstr = ".*ros-" + ros_distro + "-" + package_name.replace("_", "-") + r"_.*\.deb$"
91 print("Finding deb package: " + str(regexstr))
92
93 thisfolderfiles = []
94 for root, dirs, files in os.walk(workspace_source_folder, topdown=False):
95 thisfolderfiles = thisfolderfiles + [os.path.join(root, f) for f in files]
96
97 debianfiles = [f for f in thisfolderfiles if re.search(firstregexstr, f)]
98 print("DETECTED DEBIAN FILES:")
99 for d in debianfiles:
100 print(f"- {d}")
101
102 visited_debian_files = [
103 f for f in debianfiles if re.search(regexstr, f) and f not in already_visited
104 ]
105 debianfilename = visited_debian_files[0]
106 print("VISITED DEBIAN FILES:")
107 for d in visited_debian_files:
108 print(f"- {d}")
109
110 print("Debian file found: ")
111 print(debianfilename)
112
113 os.chdir(localpackagepath)
114 shutil.rmtree(".obj-x86_64-linux-gnu")
115 shutil.rmtree("debian")
116
117 print("installing debian package (required to build new debs): " + debianfilename)
118 installdebiantask = subprocess.Popen("sudo dpkg -i " + debianfilename, shell=True)
119 installdebiantask.wait()
120
121 return debianfilename

124def iterate_debian_generation(
125 workspace_source_folder, package_names, identified_install_packages, osversion, rosversion
126):
127 os.chdir(workspace_source_folder)
128 debianfiles = []
129 for pname in package_names:
130 try:
131 debianfiles.append(
132 build_deb_package(
133 workspace_source_folder,
134 pname,
135 identified_install_packages[pname],
136 osversion,
137 rosversion,
138 debianfiles,
139 )
140 )
141 except Exception as ex:
142 print(ex)
143 print("ERROR: package " + str(pname) + " could not be built")
144 print("existing packages: " + str(identified_install_packages.keys()))
145 raise ex
146
147 return debianfiles
