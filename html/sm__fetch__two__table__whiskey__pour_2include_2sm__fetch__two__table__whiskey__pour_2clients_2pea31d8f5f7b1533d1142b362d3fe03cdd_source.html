95 void propagateLinkStatesToTf(const gazebo_msgs::LinkStates &linksmsg, std::string linkNameFilter,
96 std::string objectPrefix, std::string globalFrame,
97 std::vector<tf::StampedTransform> &transforms)
98 {
99 transforms.clear();

101 std::vector<geometry_msgs::Pose> filteredLinkPoses;
102 for (auto &b : linksmsg.pose)
103 {
104 auto linkname = linksmsg.name[i];
105 if (linkname.find(linkNameFilter) != std::string::npos)
106 filteredLinkPoses.push_back(b);
107
108 i++;
109 }
110
111 i = 0;
112 for (auto &linkpose : filteredLinkPoses)
113 {
114 tf::Pose tfpose;
115 tf::poseMsgToTF(linkpose, tfpose);
116 tf::StampedTransform transform;
117 transform.setOrigin(tfpose.getOrigin());
118 transform.setRotation(tfpose.getRotation());
119 transform.frame_id_ = globalFrame;
120 transform.child_frame_id_ = objectPrefix + std::to_string(i + 1);
121 transform.stamp_ = ros::Time::now();
122
123 tfBroadcaster_.sendTransform(transform);
124
125 i++;
126 transforms.push_back(transform);
127 }
