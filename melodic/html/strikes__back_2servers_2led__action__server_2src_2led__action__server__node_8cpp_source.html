60void execute(const sm_dance_bot_strikes_back::LEDControlGoalConstPtr& goal) // Note: "Action" is not appended to DoDishes here
61{
62 ROS_INFO_STREAM("Tool action server request: "<< *goal);
63 cmd = *goal;
64
65 if(goal->command == sm_dance_bot_strikes_back::LEDControlGoal::CMD_ON)
66 {
67 currentState_ = sm_dance_bot_strikes_back::LEDControlResult::STATE_RUNNING;
68 }
69 else if (goal->command == sm_dance_bot_strikes_back::LEDControlGoal::CMD_OFF)
70 {
71 currentState_ = sm_dance_bot_strikes_back::LEDControlResult::STATE_IDLE;
72 }
73
74 // 10Hz internal loop
75 ros::Rate rate(20);
76
77 while(ros::ok())
78 {
79 if(as_->isPreemptRequested())
80 {
81 // a new request is being executed, we will stop this one
82 ROS_WARN("LEDActionServer request preempted. Forgetting older request.");
83 as_->setPreempted();
84 return;
85 }
86
87 publishFeedback();
88 publishStateMarker();
89 rate.sleep();
90 }
91
92 // never reach succeeded because were are interested in keeping the feedback alive
93 as_->setSucceeded();
94}

101void run()
102{
103 ros::NodeHandle n;
104 ROS_INFO("Creating tool action server");
105
106 as_ = std::make_shared<Server>(n, "led_action_server", boost::bind(&LEDActionServer::execute, this, _1), false);
107 ROS_INFO("Starting Tool Action Server");
108
109 stateMarkerPublisher_ = n.advertise<visualization_msgs::MarkerArray>("tool_markers", 1);
110
111 as_->start();

113 ros::spin();

121void publishStateMarker()
122{
123 visualization_msgs::Marker marker;

125 marker.header.stamp = ros::Time::now ();

127 marker.ns = "tool_namespace";

131
132 marker.scale.x = 0.2;
133 marker.scale.y = 0.2;
134 marker.scale.z = 0.2;
135
136 marker.color.a = 1;
137
138 if(currentState_ == sm_dance_bot_strikes_back::LEDControlResult::STATE_RUNNING)
139 {
140 // show green ball
141 marker.color.r = 0;
142 marker.color.g = 1;
143 marker.color.b = 0;
144 }
145 else if (currentState_ == sm_dance_bot_strikes_back::LEDControlResult::STATE_IDLE)
146 {
147 // show gray ball
148 marker.color.r = 0.7;
149 marker.color.g = 0.7;
150 marker.color.b = 0.7;
151 }
152 else
153 {
154 // show black ball
155 marker.color.r = 0;
156 marker.color.g = 0;
157 marker.color.b = 0;
158 }
159
160 marker.pose.orientation.w=1;
161 marker.pose.position.x=0;
162 marker.pose.position.y=0;
163 marker.pose.position.z=1;
164
165 visualization_msgs::MarkerArray ma;
166 ma.markers.push_back(marker);
167
168 stateMarkerPublisher_.publish(ma);
169}
