66void ISmaccStateMachine::updateStatusMessage()
67{
68 std::lock_guard<std::recursive_mutex> lock(m_mutex_);
69
70 if (currentStateInfo_ != nullptr)
71 {
72 ROS_WARN_STREAM("[StateMachine] setting state active "
73 << ": " << currentStateInfo_->getFullPath());
74
75 if (this->runMode_ == SMRunMode::DEBUG)
76 {
77 status_msg_.current_states.clear();
78 std::list<const SmaccStateInfo *> ancestorList;
79 currentStateInfo_->getAncestors(ancestorList);
80
81 for (auto &ancestor : ancestorList)
82 {
83 status_msg_.current_states.push_back(ancestor->toShortName());
84 }
85
86 status_msg_.global_variable_names.clear();
87 status_msg_.global_variable_values.clear();
88
89 for (auto entry : this->globalData_)
90 {
91 status_msg_.global_variable_names.push_back(entry.first);
92 status_msg_.global_variable_values.push_back(entry.second.first()); // <- invoke to_string()
93 }
94
95 status_msg_.header.stamp = ros::Time::now();
96 status_msg_.header.frame_id = "odom";
97 this->stateMachineStatusPub_.publish(status_msg_);
98 }
99 }

121void ISmaccStateMachine::initializeROS(std::string shortname)
122{
123 ROS_WARN_STREAM("State machine base creation:" << shortname);
124 // STATE MACHINE TOPICS
125 stateMachinePub_ = nh_.advertise<smacc_msgs::SmaccStateMachine>(shortname + "/smacc/state_machine_description", 1);
126 stateMachineStatusPub_ = nh_.advertise<smacc_msgs::SmaccStatus>(shortname + "/smacc/status", 1);
127 transitionLogPub_ = nh_.advertise<smacc_msgs::SmaccTransitionLogEntry>(shortname + "/smacc/transition_log", 1);
128
129 // STATE MACHINE SERVICES
130 transitionHistoryService_ = nh_.advertiseService(shortname + "/smacc/transition_log_history", &ISmaccStateMachine::getTransitionLogHistory, this);
131}
