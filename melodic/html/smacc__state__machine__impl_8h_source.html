29 template <typename TOrthogonal>
30 TOrthogonal *ISmaccStateMachine::getOrthogonal()
31 {
32 std::lock_guard<std::recursive_mutex> lock(m_mutex_);
33
34 std::string orthogonalkey = demangledTypeName<TOrthogonal>();
35 TOrthogonal *ret;
36
37 auto it = orthogonals_.find(orthogonalkey);
38
39 if (it != orthogonals_.end())
40 {
41 ROS_DEBUG("Orthogonal %s resource is being required from some state, client or component. Found resource in cache.", orthogonalkey.c_str());
42 ret = dynamic_cast<TOrthogonal *>(it->second.get());
43 return ret;
44 }
45 else
46 {
47 std::stringstream ss;
48 ss << "Orthogonal not found " << orthogonalkey.c_str() << std::endl;
49 ss << "The existing orthogonals are the following: " << std::endl;
50 for (auto &orthogonal : orthogonals_)
51 {
52 ss << " - " << orthogonal.first << std::endl;
53 }
54
55 ROS_WARN_STREAM(ss.str());
56
57 return nullptr;
58 }
59 }

62 template <typename TOrthogonal>
63 void ISmaccStateMachine::createOrthogonal()
64 {
65 this->lockStateMachine("create orthogonal");
66 std::string orthogonalkey = demangledTypeName<TOrthogonal>();
67
68 if (orthogonals_.count(orthogonalkey) == 0)
69 {
70 auto ret = std::make_shared<TOrthogonal>();
71 orthogonals_[orthogonalkey] = dynamic_pointer_cast<smacc::ISmaccOrthogonal>(ret);
72
73 ret->setStateMachine(this);
74
75 ROS_INFO("%s Orthogonal is created", orthogonalkey.c_str());
76 }
77 else
78 {
79 ROS_WARN_STREAM("There were already one existing orthogonal of type "
80 << orthogonalkey.c_str() << ". Skipping creation orthogonal request. ");
81 std::stringstream ss;
82 ss << "The existing orthogonals are the following: " << std::endl;
83 for (auto &orthogonal : orthogonals_)
84 {
85 ss << " - " << orthogonal.first << std::endl;
86 }
87 ROS_WARN_STREAM(ss.str());
88 }
89 this->unlockStateMachine("create orthogonal");
90 }

137 template <typename EventType>
138 void ISmaccStateMachine::postEvent(EventType *ev, EventLifeTime evlifetime)
139 {
140 std::lock_guard<std::recursive_mutex> guard(eventQueueMutex_);
141 if (evlifetime == EventLifeTime::CURRENT_STATE && (stateMachineCurrentAction == StateMachineInternalAction::STATE_EXITING ||
142 stateMachineCurrentAction == StateMachineInternalAction::TRANSITIONING))
143 {
144 ROS_WARN_STREAM("CURRENT STATE SCOPED EVENT SKIPPED, state is exiting/transitioning " << demangleSymbol<EventType>());
145 return;
146 // in this case we may lose/skip events, if this is not right for some cases we should create a queue
147 // to lock the events during the transitions. This issues appeared when a client asyncbehavior was posting an event
148 // meanwhile we were doing the transition, but the main thread was waiting for its correct finalization (with thread.join)
149 }
150
151 // when a postting event is requested by any component, client, or client behavior
152 // we reach this place. Now, we propagate the events to all the state state reactors to generate
153 // some more events
154
155 ROS_DEBUG_STREAM("[PostEvent entry point] " << demangleSymbol<EventType>());
156 auto currentstate = currentState_;
157 if (currentstate != nullptr)
158 {
159 propagateEventToStateReactors(currentstate, ev);
160 }
161
162 this->signalDetector_->postEvent(ev);
163 }

173 bool ISmaccStateMachine::getGlobalSMData(std::string name, T &ret)
174 {
175 std::lock_guard<std::recursive_mutex> lock(m_mutex_);
176 // ROS_WARN("get SM Data lock acquire");
177 bool success = false;
178
179 if (!globalData_.count(name))
180 {
181 // ROS_WARN("get SM Data - data do not exist");
182 success = false;
183 }
184 else
185 {
186 // ROS_WARN("get SM DAta -data exist. accessing");
187 try
188 {
189 auto &v = globalData_[name];
190
191 // ROS_WARN("get SM DAta -data exist. any cast");
192 ret = boost::any_cast<T>(v.second);
193 success = true;
194 // ROS_WARN("get SM DAta -data exist. success");
195 }
196 catch (boost::bad_any_cast &ex)
197 {
198 ROS_ERROR("bad any cast: %s", ex.what());
199 success = false;
200 }
201 }
202
203 // ROS_WARN("get SM Data lock release");
204 return success;

226 template <typename StateField, typename BehaviorType>
227 void ISmaccStateMachine::mapBehavior()
228 {
229 std::string stateFieldName = demangleSymbol(typeid(StateField).name());
230 std::string behaviorType = demangleSymbol(typeid(BehaviorType).name());
231 ROS_INFO("Mapping state field '%s' to stateReactor '%s'", stateFieldName.c_str(), behaviorType.c_str());
232 SmaccClientBehavior *globalreference;
233 if (!this->getGlobalSMData(stateFieldName, globalreference))
234 {
235 // Using the requires component approach, we force a unique existence
236 // of this component
237 BehaviorType *behavior;
238 this->requiresComponent(behavior);
239 globalreference = dynamic_cast<ISmaccClientBehavior *>(behavior);
240
241 this->setGlobalSMData(stateFieldName, globalreference);
242 }

297 template <typename TSmaccSignal, typename TMemberFunctionPrototype, typename TSmaccObjectType>
298 boost::signals2::connection ISmaccStateMachine::createSignalConnection(TSmaccSignal &signal,
299 TMemberFunctionPrototype callback,
300 TSmaccObjectType *object)
301 {
302 static_assert(std::is_base_of<ISmaccState, TSmaccObjectType>::value ||
303 std::is_base_of<ISmaccClient, TSmaccObjectType>::value ||
304 std::is_base_of<ISmaccClientBehavior, TSmaccObjectType>::value ||
305 std::is_base_of<StateReactor, TSmaccObjectType>::value ||
306 std::is_base_of<ISmaccComponent, TSmaccObjectType>::value,
307 "Only are accepted smacc types as subscribers for smacc signals");
308
309 typedef decltype(callback) ft;
310 Bind<boost::function_types::function_arity<ft>::value> binder;
311 boost::signals2::connection connection = binder.bindaux(signal, callback, object);
312
313 // long life-time objects
314 if (std::is_base_of<ISmaccComponent, TSmaccObjectType>::value ||
315 std::is_base_of<ISmaccClient, TSmaccObjectType>::value ||
316 std::is_base_of<ISmaccOrthogonal, TSmaccObjectType>::value ||
317 std::is_base_of<ISmaccStateMachine, TSmaccObjectType>::value)
318 {
319 }
320 else if (std::is_base_of<ISmaccState, TSmaccObjectType>::value ||
321 std::is_base_of<StateReactor, TSmaccObjectType>::value ||
322 std::is_base_of<ISmaccClientBehavior, TSmaccObjectType>::value)
323 {
324 ROS_INFO("[StateMachine] life-time constrained smacc signal subscription created. Subscriber is %s",
325 demangledTypeName<TSmaccObjectType>().c_str());
326 stateCallbackConnections.push_back(connection);
327 }
328 else // state life-time objects
329 {
330 ROS_WARN("[StateMachine] connecting signal to an unknown object with life-time unknown behavior. It might provoke "
331 "an exception if the object is destroyed during the execution.");
332 }
333
334 return connection;
335 }

377 template <typename StateType>
378 void ISmaccStateMachine::notifyOnStateEntryEnd(StateType *state)
379 {
380 ROS_INFO("[%s] State OnEntry code finished", demangleSymbol(typeid(StateType).name()).c_str());
381
382 for (auto pair : this->orthogonals_)
383 {
384 //ROS_INFO("ortho onentry: %s", pair.second->getName().c_str());
385 auto &orthogonal = pair.second;
386 try
387 {
388 orthogonal->onEntry();
389 }
390 catch (const std::exception &e)
391 {
392 ROS_ERROR("[Orthogonal %s] Exception on Entry - continuing with next orthogonal. Exception info: %s",
393 pair.second->getName().c_str(), e.what());
394 }
395 }
396
397 for (auto &sr : this->currentState_->getStateReactors())
398 {
399 auto srname = smacc::demangleSymbol(typeid(*sr).name()).c_str();
400 ROS_INFO("state reactor onEntry: %s", srname);
401 try
402 {
403 sr->onEntry();
404 }
405 catch (const std::exception &e)
406 {
407 ROS_ERROR("[State Reactor %s] Exception on Entry - continuing with next state reactor. Exception info: %s",
408 srname, e.what());
409 }
410 }
411
412 for (auto &eg : this->currentState_->getEventGenerators())
413 {
414 auto egname = smacc::demangleSymbol(typeid(*eg).name()).c_str();
415 ROS_INFO("state reactor onEntry: %s", egname);
416 try
417 {
418 eg->onEntry();
419 }
420 catch (const std::exception &e)
421 {
422 ROS_ERROR("[Event generator %s] Exception on Entry - continuing with next state reactor. Exception info: %s",
423 egname, e.what());
424 }
425 }
426
427 this->updateStatusMessage();
428 stateMachineCurrentAction = StateMachineInternalAction::STATE_STEADY;
429 }

452 template <typename StateType>
453 void ISmaccStateMachine::notifyOnStateExitting(StateType *state)
454 {
455 stateMachineCurrentAction = StateMachineInternalAction::STATE_EXITING;
456
457 auto fullname = demangleSymbol(typeid(StateType).name());
458 ROS_WARN_STREAM("exiting state: " << fullname);
459 //this->setParam("destroyed", true);
460
461 ROS_INFO_STREAM("Notification State Exit: leaving state" << state);
462 for (auto pair : this->orthogonals_)
463 {
464 auto &orthogonal = pair.second;
465 try
466 {
467 orthogonal->onExit();
468 }
469 catch (const std::exception &e)
470 {
471 ROS_ERROR("[Orthogonal %s] Exception onExit - continuing with next orthogonal. Exception info: %s",
472 pair.second->getName().c_str(), e.what());
473 }
474 }
475
476 for (auto &sr : state->getStateReactors())
477 {
478 auto srname = smacc::demangleSymbol(typeid(*sr).name()).c_str();
479 ROS_INFO("state reactor OnExit: %s", srname);
480 try
481 {
482 sr->onExit();
483 }
484 catch (const std::exception &e)
485 {
486 ROS_ERROR("[State Reactor %s] Exception on OnExit - continuing with next state reactor. Exception info: %s",
487 srname, e.what());
488 }
489 }
490
491 for (auto &eg : state->getEventGenerators())
492 {
493 auto egname = smacc::demangleSymbol(typeid(*eg).name()).c_str();
494 ROS_INFO("state reactor OnExit: %s", egname);
495 try
496 {
497 eg->onExit();
498 }
499 catch (const std::exception &e)
500 {
501 ROS_ERROR("[State Reactor %s] Exception on OnExit - continuing with next state reactor. Exception info: %s",
502 egname, e.what());
503 }
504 }
505
506 this->lockStateMachine("state exit");
507
508 for (auto &conn : this->stateCallbackConnections)
509 {
510 ROS_WARN_STREAM("[StateMachine] Disconnecting scoped-lifetime SmaccSignal subscription");
511 conn.disconnect();
512 }
513
514 this->stateCallbackConnections.clear();
515
516 currentState_ = nullptr;
517 }
