166 void computePregraspPoseFromCubePose(geometry_msgs::PoseStamped &objectPose)
167 {
168 objectPose.pose.position.x -= 0;
169 objectPose.pose.position.z += 0.3;
170
171 // ------ cube grasping orientation -------
172 // grasp the object with the gripper using top-to-bottom direction
173 auto cubeYawOnTable = tf::getYaw(objectPose.pose.orientation);
174
175 const double degrees90 = M_PI / 2;
176 while (cubeYawOnTable > degrees90)
177 {
178 cubeYawOnTable -= degrees90;
179 }
180
181 while (cubeYawOnTable < -degrees90)
182 {
183 cubeYawOnTable += degrees90;
184 }
185
186 ROS_INFO("cube yaw: %lf", cubeYawOnTable);
187 auto quat = tf::createQuaternionFromRPY(0, M_PI / 2, cubeYawOnTable);
188 tf::quaternionTFToMsg(quat, objectPose.pose.orientation);
189 }
